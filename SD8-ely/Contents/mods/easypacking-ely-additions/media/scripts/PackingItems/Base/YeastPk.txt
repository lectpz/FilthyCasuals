module Packing
{
    imports
    {
        Base,
    }

    item 5pkYeast
    {
        DisplayName         = 5 x Yeast,
        DisplayCategory     = Food,
        Weight              = 3,
        Type                = Normal,
        Icon                = Yeast,
        WorldStaticModel    = Yeast,
    }

    item 10pkYeast
    {
        DisplayName         = 10 x Yeast,
        DisplayCategory     = Food,
        Weight              = 5,
        Type                = Normal,
        Icon                = Yeast,
        WorldStaticModel    = Yeast
    }

    recipe Pack 5
    {
        destroy Yeast=5,

        Result  :5pkYeast,
        OnCreate:Recipe.OnCreate.SaveFood
        OnTest  :Recipe.OnTest.IsFavorite,
        Time    :50.0,
        Category:Storage,
    }

    recipe Unpack 5
    {
        destroy 5pkYeast,

        Result          :Yeast=5,
        OnCreate        :Recipe.OnCreate.LoadFood,
        OnGiveXP        :Recipe.OnGiveXP.None,
        RemoveResultItem:true,
        Time            :50.0,
        Category        :Storage
    }

    recipe Pack 10
    {
        destroy Yeast=10,

        Result  :10pkYeast,
        OnCreate:Recipe.OnCreate.SaveFood
        OnTest  :Recipe.OnTest.IsFavorite,
        Time    :75.0,
        Category:Storage,
    }

    recipe Unpack 10
    {
        destroy 10pkYeast,

        Result          :Yeast=10,
        OnCreate        :Recipe.OnCreate.LoadFood,
        OnGiveXP        :Recipe.OnGiveXP.None,
        RemoveResultItem:true,
        Time            :50.0,
        Category        :Storage
    }

    recipe Put Together
    {
        destroy 5pkYeast=2,
        
        Result  :10pkYeast,
        Time    :50.0
    }

    recipe Split in Two
    {
        destroy 10pkYeast,

        CanBeDoneFromFloor  :true,
        Result              :5pkYeast,
        Time                :50.0
    }
